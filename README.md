## Install

```bash
npm i tsdown
```

## Configuration

```ts
// tsdown.config.ts
import { defineConfig } from 'tsdown'

export default defineConfig({
  entry: ['./src'],
  // ...
})
```

## Usage

```bash
npx tsdown
```

## Migrate from tsup

```bash
npx tsdown migrate
```

Please make sure to commit your changes before migrating.
